```python
from fastapi import UploadFile, File

@produtos_router.post("/{produto_id}/foto")
async def upload_foto(produto_id: str, file: UploadFile = File(...)):
    path = f"produtos/{produto_id}/{file.filename}"
    content = await file.read()
    get_client().storage.from_("product-images").upload(path, content, {"content-type": file.content_type})
    url = get_client().storage.from_("product-images").create_signed_url(path, 86400)
    get_client().table("produtos").update({"foto_url": url["signedURL"]}).eq("id", produto_id).execute()
    return {"foto_url": url["signedURL"]}
```
