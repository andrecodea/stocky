# app/routers/estoque.py
from fastapi import APIRouter, Depends
from app.database import supabase_client
from app.dependencies import get_current_user

router = APIRouter(prefix="/estoque", tags=["estoque"])

@router.get("/alertas")
async def alertas_esgotamento(usuario=Depends(get_current_user)):
    prods = supabase_client.table("produtos").select(
        "id, nome, estoque_minimo, estoque_atual(quantidade_atual)"
    ).execute()

    alertas = []
    for p in prods.data:
        qtd = (p.get("estoque_atual") or [{}])[0].get("quantidade_atual", 0) or 0
        if qtd < p["estoque_minimo"]:
            alertas.append({
                "id": p["id"],
                "nome": p["nome"],
                "estoque_minimo": p["estoque_minimo"],
                "quantidade_atual": qtd,
                "diferenca": p["estoque_minimo"] - qtd,
            })
    return alertas
```

# em app/routers/produtos.py
from fastapi import UploadFile, File
from uuid import UUID

@router.post("/{produto_id}/foto")
async def upload_foto(produto_id: UUID, file: UploadFile = File(...), usuario=Depends(get_current_user)):
    path = f"produtos/{produto_id}/{file.filename}"
    content = await file.read()
    supabase_client.storage.from_("product-images").upload(path, content, {"content-type": file.content_type})
    url = supabase_client.storage.from_("product-images").create_signed_url(path, 86400)
    supabase_client.table("produtos").update({"foto_url": url["signedURL"]}).eq("id", str(produto_id)).execute()
    return {"foto_url": url["signedURL"]}
```
