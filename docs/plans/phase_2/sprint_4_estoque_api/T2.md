@produtos_router.get("/")
def get_produtos():

@produtos_router.get("/{produto_id}")
def get_produto(produto_id: str):
    produto = buscar_produto(produto_id)
    if not produto:
        raise HTTPException(status_code=404, detail="Produto não encontrado")
    return produto

```python
@produtos_router.post("/", status_code=status.HTTP_201_CREATED)
def post_produto(data: ProductCreate):
    return criar_produto(data)

@produtos_router.put("/{produto_id}")
def put_produto(produto_id: str, data: ProductUpdate):
    produto = atualizar_produto(produto_id, data)
    if not produto:
        raise HTTPException(status_code=404, detail="Produto não encontrado")
    return produto

@produtos_router.delete("/{produto_id}", status_code=status.HTTP_204_NO_CONTENT)
def delete_produto(produto_id: str):
    if not deletar_produto(produto_id):
        raise HTTPException(status_code=404, detail="Produto não encontrado")
```
