# app/models/produto.py
from pydantic import BaseModel
from typing import Optional
from uuid import UUID
from datetime import datetime

class ProdutoCreate(BaseModel):
    nome: str
    sku: Optional[str] = None
    preco_custo: float = 0
    preco_venda: float = 0
    unidade: str = "un"
    estoque_minimo: int = 0

class ProdutoResponse(ProdutoCreate):
    id: UUID
    foto_url: Optional[str] = None
    quantidade_atual: Optional[int] = 0
    criado_em: datetime

class MovimentacaoCreate(BaseModel):
    produto_id: UUID
    tipo: str  # "entrada" | "saida" | "ajuste"
    quantidade: int
    lote_id: Optional[UUID] = None
    observacao: Optional[str] = None
```

# app/routers/produtos.py
from fastapi import APIRouter, Depends, HTTPException, status
from app.database import supabase_client
from app.dependencies import get_current_user
from app.models.usuario import UsuarioAtual
from app.models.produto import ProdutoCreate, MovimentacaoCreate

router = APIRouter(prefix="/produtos", tags=["produtos"])

@router.get("/")
async def listar_produtos(usuario: UsuarioAtual = Depends(get_current_user)):
    result = supabase_client.table("produtos").select("*, estoque_atual(quantidade_atual)").execute()
    return result.data

@router.post("/", status_code=status.HTTP_201_CREATED)
async def criar_produto(produto: ProdutoCreate, usuario: UsuarioAtual = Depends(get_current_user)):
    if usuario.role != "admin":
        raise HTTPException(status_code=403, detail="Apenas admin pode criar produtos")
    result = supabase_client.table("produtos").insert(produto.model_dump()).execute()
    return result.data[0]

@router.post("/movimentacoes", status_code=status.HTTP_201_CREATED)
async def criar_movimentacao(mov: MovimentacaoCreate, usuario: UsuarioAtual = Depends(get_current_user)):
    data = mov.model_dump()
    data["usuario_id"] = usuario.id
    result = supabase_client.table("movimentacoes").insert(data).execute()
    return result.data[0]
```
