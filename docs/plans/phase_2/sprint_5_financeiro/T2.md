```python
# app/routers/financeiro.py
from fastapi import APIRouter, Depends
from app.database import supabase_client
from app.dependencies import get_current_user
from app.models.financeiro import VendaCreate

router = APIRouter(prefix="/financeiro", tags=["financeiro"])

@router.post("/vendas", status_code=201)
async def registrar_venda(venda: VendaCreate, usuario=Depends(get_current_user)):
    total = sum(i.quantidade * i.preco_unitario for i in venda.itens)
    v = supabase_client.table("vendas").insert({"usuario_id": usuario.id, "total": total}).execute()
    venda_id = v.data[0]["id"]
    itens = [{"venda_id": venda_id, **i.model_dump()} for i in venda.itens]
    supabase_client.table("itens_venda").insert(itens).execute()
    return {"id": venda_id, "total": total, "itens": len(itens)}
```

```python
@router.get("/margem")
async def margem_por_produto(usuario=Depends(get_current_user)):
    result = supabase_client.table("margem_produto").select("*").execute()
    return result.data

@router.get("/resumo")
async def resumo_financeiro(usuario=Depends(get_current_user)):
    result = supabase_client.table("margem_produto").select("receita_total, custo_total, margem_bruta").execute()
    rows = result.data
    return {
        "receita_total": sum(r["receita_total"] or 0 for r in rows),
        "custo_total": sum(r["custo_total"] or 0 for r in rows),
        "margem_bruta": sum(r["margem_bruta"] or 0 for r in rows),
    }
```
