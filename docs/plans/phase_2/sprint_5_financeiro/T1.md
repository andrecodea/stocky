```sql
CREATE TABLE IF NOT EXISTS vendas (
  id          uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  usuario_id  uuid REFERENCES auth.users(id),
  total       numeric(10,2) NOT NULL DEFAULT 0,
  criado_em   timestamptz NOT NULL DEFAULT now()
);

CREATE TABLE IF NOT EXISTS itens_venda (
  id               uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  venda_id         uuid NOT NULL REFERENCES vendas(id) ON DELETE CASCADE,
  produto_id       uuid NOT NULL REFERENCES produtos(id),
  quantidade       integer NOT NULL,
  preco_unitario   numeric(10,2) NOT NULL,
  custo_unitario   numeric(10,2) NOT NULL DEFAULT 0
);
```

<details>
<summary>💡 Solução — Passo 2</summary>

```sql
CREATE OR REPLACE VIEW margem_produto AS
SELECT
  iv.produto_id,
  p.nome,
  SUM(iv.quantidade * iv.preco_unitario)   AS receita_total,
  SUM(iv.quantidade * iv.custo_unitario)   AS custo_total,
  SUM(iv.quantidade * (iv.preco_unitario - iv.custo_unitario)) AS margem_bruta,
  ROUND(
    SUM(iv.quantidade * (iv.preco_unitario - iv.custo_unitario)) /
    NULLIF(SUM(iv.quantidade * iv.preco_unitario), 0) * 100, 2
  ) AS margem_pct
FROM itens_venda iv
JOIN produtos p ON p.id = iv.produto_id
GROUP BY iv.produto_id, p.nome;
```

```sql
ALTER TABLE vendas ENABLE ROW LEVEL SECURITY;
ALTER TABLE itens_venda ENABLE ROW LEVEL SECURITY;

CREATE POLICY "vendas_select" ON vendas FOR SELECT TO authenticated
  USING (usuario_id = auth.uid() OR is_admin());
CREATE POLICY "vendas_insert" ON vendas FOR INSERT TO authenticated
  WITH CHECK (usuario_id = auth.uid());
CREATE POLICY "vendas_admin_delete" ON vendas FOR DELETE TO authenticated
  USING (is_admin());

CREATE POLICY "itens_select" ON itens_venda FOR SELECT TO authenticated
  USING (EXISTS (SELECT 1 FROM vendas v WHERE v.id = venda_id AND (v.usuario_id = auth.uid() OR is_admin())));
CREATE POLICY "itens_insert" ON itens_venda FOR INSERT TO authenticated WITH CHECK (true);

CREATE INDEX IF NOT EXISTS idx_itens_venda_produto_id ON itens_venda(produto_id);
CREATE INDEX IF NOT EXISTS idx_vendas_criado_em ON vendas(criado_em DESC);
```
