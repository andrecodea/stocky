from langchain_core.messages import HumanMessage
from typing import Optional

def identificar_produto_por_foto(base64_image: str, mime_type: str = "image/jpeg") -> ProdutoIdentificado:
    llm = ChatOpenAI(
        base_url="https://openrouter.ai/api/v1",
        api_key=settings.openrouter_api_key,
        model=settings.openrouter_model_vision,
    )
    structured_llm = llm.with_structured_output(ProdutoIdentificado)

    message = HumanMessage(content=[
        {"type": "text", "text": (
            "Analise esta imagem de produto. Extraia: nome do produto, código/SKU (se visível), "
            "descrição breve e sua confiança na identificação (0.0 a 1.0)."
        )},
        {"type": "image_url", "image_url": {"url": f"data:{mime_type};base64,{base64_image}"}},
    ])

    return structured_llm.invoke([message])
```

import base64
from fastapi import UploadFile, File

@router.post("/identificar-produto")
async def identificar_produto(
    file: UploadFile = File(...),
    usuario=Depends(get_current_user)
):
    content = await file.read()
    b64 = base64.b64encode(content).decode("utf-8")
    return identificar_produto_por_foto(b64, file.content_type or "image/jpeg")
```
