```python
import sys
from pathlib import Path

from services.ai_service import recommend_restock, answer_query
```

```python
from pydantic import BaseModel
from typing import Optional

class ChatRequest(BaseModel):
    query: str
    historico: list[dict] = []

@ia_router.post("/chat")
def chat(body: ChatRequest):
    estoque = listar_estoque_atual()
    resposta = answer_query(
        query=body.query,
        inventory=estoque,
        history=body.historico,
    )
    return {"resposta": resposta}
```

```python
import base64
from fastapi import UploadFile, File
from langchain_openai import ChatOpenAI
from langchain_core.messages import HumanMessage
from pydantic import BaseModel
from config import settings

class ProdutoIdentificado(BaseModel):
    nome: str
    sku: Optional[str] = None
    descricao: Optional[str] = None
    confianca: float

@ia_router.post("/identificar-produto")
async def identificar_produto(file: UploadFile = File(...)):
    content = await file.read()
    b64 = base64.b64encode(content).decode()
    llm = ChatOpenAI(
        base_url="https://openrouter.ai/api/v1",
        api_key=settings.openrouter_api_key,
        model=settings.openrouter_model_vision,
    ).with_structured_output(ProdutoIdentificado)
    msg = HumanMessage(content=[
        {"type": "text", "text": "Analise esta imagem e extraia: nome do produto, SKU (se visível), descrição breve e confiança (0.0 a 1.0)."},
        {"type": "image_url", "image_url": {"url": f"data:{file.content_type};base64,{b64}"}},
    ])
    return llm.invoke([msg])
```
