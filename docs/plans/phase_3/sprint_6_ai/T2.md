### Conceito: Structured Output com LangChain

> `llm.with_structured_output(PydanticModel)` instrui o modelo a retornar JSON válido que o LangChain valida contra o schema Pydantic.  
> Sem isso, o output é texto livre e o frontend não consegue processar.

```python
import json
from langchain_openai import ChatOpenAI
from langchain_core.prompts import ChatPromptTemplate
from app.config import settings

def gerar_recomendacoes(estoque_data: list) -> Recomendacoes:
    llm = ChatOpenAI(
        base_url="https://openrouter.ai/api/v1",
        api_key=settings.openrouter_api_key,
        model=settings.openrouter_model_chat,
    )
    structured_llm = llm.with_structured_output(Recomendacoes)

    prompt = ChatPromptTemplate.from_messages([
        ("system", "Você é um especialista em gestão de estoque para pequenos negócios. "
                   "Analise os dados de estoque abaixo e recomende reposição para produtos "
                   "que estão abaixo ou próximos do estoque mínimo. Seja objetivo e prático."),
        ("human", "Dados de estoque:\n{estoque_json}\n\nGere recomendações de reposição."),
    ])

    chain = prompt | structured_llm
    return chain.invoke({"estoque_json": json.dumps(estoque_data, ensure_ascii=False, default=str)})
```

```python
# app/routers/ia.py
from fastapi import APIRouter, Depends
from app.database import supabase_client
from app.dependencies import get_current_user
from app.services.ai_service import gerar_recomendacoes

router = APIRouter(prefix="/ia", tags=["ia"])

@router.post("/recomendar-reposicao")
async def recomendar_reposicao(usuario=Depends(get_current_user)):
    result = supabase_client.table("produtos").select(
        "id, nome, estoque_minimo, estoque_atual(quantidade_atual)"
    ).execute()
    return gerar_recomendacoes(result.data)
```
