```typescript
// mobile/src/screens/EstoqueScreen.tsx
import React from 'react';
import { FlatList, Text, View, StyleSheet } from 'react-native';
import { Card } from '../components/Card';
import { Badge } from '../components/Badge';
import { MOCK_PRODUTOS } from '../mock/data';
import { theme } from '../theme';

const getVariant = (qtd: number, min: number) =>
  qtd === 0 ? 'danger' : qtd < min ? 'warning' : 'ok';

export const EstoqueScreen = () => (
  <View style={styles.container}>
    <Text style={styles.header}>Estoque ({MOCK_PRODUTOS.length} produtos)</Text>
    <FlatList
      data={MOCK_PRODUTOS}
      keyExtractor={p => p.id}
      renderItem={({ item }) => (
        <Card style={styles.item}>
          <View style={styles.row}>
            <View style={{ flex: 1 }}>
              <Text style={styles.nome}>{item.nome}</Text>
              <Text style={styles.sku}>{item.sku}</Text>
            </View>
            <Badge
              label={`${item.quantidade_atual} ${item.unidade}`}
              variant={getVariant(item.quantidade_atual, item.estoque_minimo)}
            />
          </View>
        </Card>
      )}
    />
  </View>
);

const styles = StyleSheet.create({
  container: { flex: 1, backgroundColor: theme.colors.background, padding: theme.spacing.md },
  header: { fontSize: theme.fontSize.xl, fontWeight: '700', marginBottom: theme.spacing.md, color: theme.colors.text },
  item: { marginBottom: theme.spacing.sm },
  row: { flexDirection: 'row', alignItems: 'center' },
  nome: { fontSize: theme.fontSize.md, fontWeight: '600', color: theme.colors.text },
  sku: { fontSize: theme.fontSize.sm, color: theme.colors.textMuted, marginTop: 2 },
});
```

```typescript
// mobile/src/screens/AlertasScreen.tsx
import React from 'react';
import { FlatList, Text, View, StyleSheet } from 'react-native';
import { Card } from '../components/Card';
import { MOCK_ALERTAS } from '../mock/data';
import { theme } from '../theme';

export const AlertasScreen = () => (
  <View style={styles.container}>
    <Text style={styles.header}>Alertas ({MOCK_ALERTAS.length})</Text>
    {MOCK_ALERTAS.length === 0 ? (
      <Text style={styles.ok}>Estoque saudável ✓</Text>
    ) : (
      <FlatList
        data={MOCK_ALERTAS}
        keyExtractor={p => p.id}
        renderItem={({ item }) => (
          <Card style={styles.alerta}>
            <Text style={styles.nome}>{item.nome}</Text>
            <Text style={styles.detalhe}>Atual: {item.quantidade_atual} | Mínimo: {item.estoque_minimo}</Text>
            <Text style={styles.diff}>Faltam {item.estoque_minimo - item.quantidade_atual} unidades</Text>
          </Card>
        )}
      />
    )}
  </View>
);

const styles = StyleSheet.create({
  container: { flex: 1, backgroundColor: theme.colors.background, padding: theme.spacing.md },
  header: { fontSize: theme.fontSize.xl, fontWeight: '700', marginBottom: theme.spacing.md, color: theme.colors.danger },
  ok: { fontSize: theme.fontSize.lg, color: theme.colors.success, textAlign: 'center', marginTop: 40 },
  alerta: { marginBottom: theme.spacing.sm, borderLeftWidth: 4, borderLeftColor: theme.colors.danger },
  nome: { fontSize: theme.fontSize.md, fontWeight: '600', color: theme.colors.text },
  detalhe: { fontSize: theme.fontSize.sm, color: theme.colors.textMuted, marginTop: 4 },
  diff: { fontSize: theme.fontSize.sm, color: theme.colors.danger, fontWeight: '600', marginTop: 4 },
});
```
