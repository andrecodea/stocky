```typescript
// mobile/src/components/Badge.tsx
import React from 'react';
import { Text, View, StyleSheet } from 'react-native';
import { theme } from '../theme';

const variantColors = {
  ok: theme.colors.success,
  warning: theme.colors.warning,
  danger: theme.colors.danger,
};

interface BadgeProps { label: string; variant: 'ok' | 'warning' | 'danger'; }

export const Badge: React.FC<BadgeProps> = ({ label, variant }) => (
  <View style={[styles.badge, { backgroundColor: variantColors[variant] + '20', borderColor: variantColors[variant] }]}>
    <Text style={[styles.text, { color: variantColors[variant] }]}>{label}</Text>
  </View>
);

const styles = StyleSheet.create({
  badge: { paddingHorizontal: 8, paddingVertical: 2, borderRadius: theme.borderRadius.sm, borderWidth: 1 },
  text: { fontSize: theme.fontSize.sm, fontWeight: '600' },
});
```

```typescript
// mobile/src/navigation/AppNavigator.tsx
import React from 'react';
import { createBottomTabNavigator } from '@react-navigation/bottom-tabs';
import { Text, View } from 'react-native';
import { theme } from '../theme';

const Tab = createBottomTabNavigator();

const Placeholder = (label: string) => () => (
  <View style={{ flex: 1, alignItems: 'center', justifyContent: 'center' }}>
    <Text>{label}</Text>
  </View>
);

export const AppNavigator = () => (
  <Tab.Navigator screenOptions={{ tabBarActiveTintColor: theme.colors.primary }}>
    <Tab.Screen name="Estoque" component={Placeholder('Estoque')} />
    <Tab.Screen name="Alertas" component={Placeholder('Alertas')} />
    <Tab.Screen name="IA" component={Placeholder('IA')} />
  </Tab.Navigator>
);
```
