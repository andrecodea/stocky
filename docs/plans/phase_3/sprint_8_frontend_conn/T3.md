```typescript
// mobile/src/screens/IAScreen.tsx
import React, { useState } from 'react';
import { View, Text, TouchableOpacity, FlatList, ActivityIndicator, StyleSheet } from 'react-native';
import { api } from '../lib/api';
import { Card } from '../components/Card';
import { theme } from '../theme';

interface Recomendacao { produto_id: string; nome: string; quantidade_sugerida: number; motivo: string; }
interface Resultado { itens: Recomendacao[]; resumo: string; }

export const IAScreen = () => {
  const [resultado, setResultado] = useState<Resultado | null>(null);
  const [loading, setLoading] = useState(false);

  const gerar = async () => {
    setLoading(true);
    try {
      const r = await api.post('/ia/recomendar-reposicao');
      setResultado(r.data);
    } finally {
      setLoading(false);
    }
  };

  return (
    <View style={styles.container}>
      <Text style={styles.header}>Assistente IA</Text>
      <TouchableOpacity style={styles.button} onPress={gerar} disabled={loading}>
        <Text style={styles.buttonText}>{loading ? 'Analisando estoque...' : 'Gerar Recomendações'}</Text>
      </TouchableOpacity>
      {loading && <ActivityIndicator style={{ marginTop: 20 }} color={theme.colors.primary} size="large" />}
      {resultado && (
        <>
          <Card style={styles.resumo}>
            <Text style={styles.resumoText}>{resultado.resumo}</Text>
          </Card>
          <FlatList
            data={resultado.itens}
            keyExtractor={i => i.produto_id}
            renderItem={({ item }) => (
              <Card style={styles.item}>
                <Text style={styles.nome}>{item.nome}</Text>
                <Text style={styles.qtd}>Repor: {item.quantidade_sugerida} unidades</Text>
                <Text style={styles.motivo}>{item.motivo}</Text>
              </Card>
            )}
          />
        </>
      )}
    </View>
  );
};

const styles = StyleSheet.create({
  container: { flex: 1, backgroundColor: theme.colors.background, padding: theme.spacing.md },
  header: { fontSize: theme.fontSize.xl, fontWeight: '700', color: theme.colors.text, marginBottom: theme.spacing.md },
  button: { backgroundColor: theme.colors.primary, borderRadius: theme.borderRadius.md, padding: theme.spacing.md, alignItems: 'center' },
  buttonText: { color: '#fff', fontSize: theme.fontSize.lg, fontWeight: '700' },
  resumo: { marginTop: theme.spacing.md, backgroundColor: theme.colors.primary + '10', borderLeftWidth: 4, borderLeftColor: theme.colors.primary },
  resumoText: { fontSize: theme.fontSize.md, color: theme.colors.text, lineHeight: 22 },
  item: { marginTop: theme.spacing.sm },
  nome: { fontSize: theme.fontSize.md, fontWeight: '600', color: theme.colors.text },
  qtd: { fontSize: theme.fontSize.sm, color: theme.colors.primary, fontWeight: '700', marginTop: 4 },
  motivo: { fontSize: theme.fontSize.sm, color: theme.colors.textMuted, marginTop: 4 },
});
```
