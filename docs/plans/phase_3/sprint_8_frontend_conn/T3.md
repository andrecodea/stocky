## Para React Native (pós-POC)

```typescript
// IAScreen — POST /ia/recomendar-reposicao
const recomendar = async () => {
  setLoading(true);
  const r = await api.post('/ia/recomendar-reposicao');
  setResultado(r.data);
  setLoading(false);
};

// ChatScreen — POST /ia/chat
const enviar = async (query: string) => {
  const r = await api.post('/ia/chat', { query, historico });
  setHistorico(prev => [...prev,
    { role: 'user', content: query },
    { role: 'assistant', content: r.data.resposta },
  ]);
};
```
